   40 viewRootHandler :: API -> [(BS.ByteString, BS.ByteString)] -> Action
   41 viewRootHandler api _ = -- TOOD: ensure GET
   42   withAuth $ do
   43     down <- peeks serviceDown
   44     when (api == HTML && isNothing down) angular
   45     case api of
   46       JSON -> return $ okResponse [] JSON.emptyObject
   47       HTML -> peeks $ okResponse [] . maybe htmlRoot htmlDown down
