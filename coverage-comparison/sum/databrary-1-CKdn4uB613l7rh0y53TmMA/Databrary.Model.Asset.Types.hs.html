   84 blankAsset vol = Asset
   85   { assetRow = AssetRow
   86     { assetId = error "blankAsset"
   87     , assetFormat = unknownFormat
   88     , assetRelease = Nothing
   89     , assetName = Nothing
   90     , assetDuration = Nothing
   91     , assetSHA1 = Nothing
