   34 postComment :: ActionRoute (Id Slot)
   35 postComment = action POST (pathJSON >/> pathSlotId </< "comment") $ \si -> withAuth $ do
   36   u <- authAccount
   37   s <- getSlot PermissionSHARED Nothing si
   38   (c, p) <- runForm (Nothing :: Maybe (RequestContext -> FormHtml a)) $ do
   39     csrfForm
   40     text <- "text" .:> (deformRequired =<< deform)
   41     parent <- "parent" .:> deformNonEmpty (deformMaybe' "comment not found" =<< lift . lookupComment =<< deform)
   42     let _ = CreateOrUpdateCommentRequest text (fmap commentId parent)
   43     return ((blankComment u s)
   44       { commentText = text
   45       , commentParents = maybe [] (return . commentId) parent
   46       }, parent)
   47   c' <- addComment c
   48   top <- containerIsVolumeTop (slotContainer s)
   49   forM_ p $ \r -> when (on (/=) (partyId . partyRow . accountParty) (commentWho r) u) $
   50     createNotification (blankNotification (commentWho r) NoticeCommentReply)
   51       { notificationContainerId = top `unlessUse` ((containerId . containerRow . slotContainer . commentSlot) c')
   52       , notificationSegment = Just $ (slotSegment . commentSlot) c'
   53       , notificationCommentId = Just $ commentId c'
   54       }
   55   createVolumeNotification ((containerVolume . slotContainer . commentSlot) c') $ \n -> (n NoticeCommentVolume)
   56     { notificationContainerId = top `unlessUse` ((containerId . containerRow . slotContainer . commentSlot) c')
   57     , notificationSegment = Just $ (slotSegment . commentSlot) c'
   58     , notificationCommentId = Just $ commentId c'
   59     }
   60   return $ okResponse [] $ JSON.recordEncoding $ commentJSON c'
   61   -- HTML -> peeks $ otherRouteResponse [] (viewSlot False) (api, (Just (view c'), slotId (commentSlot c')))
