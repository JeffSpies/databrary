   26 data QueryFundersRequest = QueryFundersRequest T.Text Bool
   27 
   28 queryFunderHandler :: Action -- TODO: GET only
   29 queryFunderHandler = withAuth $ do
   30   _ <- authAccount
   31   QueryFundersRequest q a <- runForm Nothing $ liftM2 QueryFundersRequest
   32     ("query" .:> (deformRequired =<< deform))
   33     ("all" .:> deform)
   34   r <- if a
   35     then focusIO $ searchFundRef q
   36     else findFunders q
   37   return $ okResponse [] $ JSON.mapObjects funderJSON r

   39 data CreateOrUpdateVolumeFundingRequest =
   40     CreateOrUpdateVolumeFundingRequest [T.Text]
   41 
   42 postVolumeFunding :: ActionRoute (Id Volume, Id Funder)
   43 postVolumeFunding = action POST (pathJSON >/> pathId </> pathId) $ \(vi, fi) -> withAuth $ do
   44   v <- getVolume PermissionEDIT vi
   45   f <- maybeAction =<< lookupFunderRef fi
   46   CreateOrUpdateVolumeFundingRequest a <- runForm Nothing $ do
   47     csrfForm
   48     CreateOrUpdateVolumeFundingRequest <$> ("awards" .:> filter (not . T.null) <$> withSubDeforms (\_ -> deform))
   49   let fa = Funding f a
   50   _ <- changeVolumeFunding v fa
   51   return $ okResponse [] $ JSON.pairs $ fundingJSON fa
   52 
   53 deleteVolumeFunder :: ActionRoute (Id Volume, Id Funder)
   54 deleteVolumeFunder = action DELETE (pathJSON >/> pathId </> pathId) $ \(vi, fi) -> withAuth $ do
   55   guardVerfHeader
   56   v <- getVolume PermissionEDIT vi
   57   _ <- removeVolumeFunder v fi
   58   return $ okResponse [] $ JSON.recordEncoding $ volumeJSONSimple v
