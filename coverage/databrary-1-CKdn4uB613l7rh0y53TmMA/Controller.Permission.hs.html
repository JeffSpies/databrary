   22 -- logic inside of checkPermission and checkDataPermission should be inside of model layer

   24 checkPermission :: Has Permission a => Permission -> a -> Handler a
   25 checkPermission requiredPermissionLevel objectWithCurrentUserPermLevel =
   26   checkPermission2 view requiredPermissionLevel objectWithCurrentUserPermLevel
   27 
   28 checkPermission2 :: (a -> Permission) -> Permission -> a -> Handler a
   29 checkPermission2 getCurrentUserPermLevel requestingAccessAtPermLevel obj = do
   30   unless (getCurrentUserPermLevel obj >= requestingAccessAtPermLevel) $ do
   31     resp <- peeks (\reqCtxt -> forbiddenResponse reqCtxt)
   32     result resp
   33   return obj

   35 userCanReadData :: (a -> EffectiveRelease) -> (a -> VolumeRolePolicy) -> a -> Handler a
   36 userCanReadData getObjEffectiveRelease getCurrentUserPermLevel obj = do
   37   unless (canReadData2 getObjEffectiveRelease getCurrentUserPermLevel obj) $ do
   38     resp <- peeks (\reqCtxt -> forbiddenResponse reqCtxt)
   39     result resp
   40   return obj

   42 -- |
   43 -- Pulls the Account out of the Handler context
   44 authAccount :: Handler Account
   45 authAccount = do
   46   ident <- peek
   47   case ident of
   48     NotLoggedIn -> result =<< peeks forbiddenResponse
   49     IdentityNotNeeded -> result =<< peeks forbiddenResponse
   50     Identified s -> return $ view s
   51     ReIdentified u -> return $ view u

   71 --     c. 1 constructor has a concrete SiteAuth field
   72 -- 2. It "inherits" the (Has Access) of the SiteAuth

   76 -- | (Maybe) tests whether someone is a superadmin?
   77 checkMemberADMIN :: Handler ()
   78 checkMemberADMIN = do
   79   a :: Access <- peek
   80   let admin = accessMember' a
   81   void $ checkPermission PermissionADMIN admin

   83 checkVerfHeader :: Handler Bool
   84 checkVerfHeader = do
   85   header <- peeks $ lookupRequestHeader "x-csverf"
   86   peeks $ or . liftM2 (==) header . identityVerf
   87 
   88 guardVerfHeader :: Handler ()
   89 guardVerfHeader = do
   90   c <- checkVerfHeader
   91   unless c $ result =<< peeks forbiddenResponse
