   40 -- | This is the context for when you don't have an identity, but you have a
   41 -- fully initialized, "command line" access to the system.
   42 data ActionContext = ActionContext
   43   { contextService :: !Service -- ^ All initialized services; "the imperative shell"
   44   , contextTimestamp :: !Timestamp -- ^ When the ActionContextM action is running (i.e., NOW)
   45   , contextResourceState :: !InternalState -- ^ Optimization for MonadResource
   46   , contextDB :: !DBConn -- ^ The specific connection chosen for the running action?
   47   }
   48 
   49 instance Has Service ActionContext where
   50   view = contextService
   51 instance Has Service.Notification.Notifications ActionContext where
   52    view = (view . contextService)
   53 instance Has Solr.Service.Solr ActionContext where
   54   view = (view . contextService)
   55 instance Has Ingest.Service.Ingest ActionContext where
   56   view = (view . contextService)
   57 instance Has Static.Service.Static ActionContext where
   58   view = (view . contextService)
   59 instance Has HTTP.Client.HTTPClient ActionContext where
   60   view = (view . contextService)
   61 instance Has Web.Types.Web ActionContext where
   62   view = (view . contextService)
   63 instance Has Store.AV.AV ActionContext where
   64   view = (view . contextService)
   65 instance Has Store.Types.Storage ActionContext where
   66   view = (view . contextService)
   67 instance Has Service.Messages.Messages ActionContext where
   68   view = (view . contextService)
   69 instance Has Service.Log.Logs ActionContext where
   70   view = (view . contextService)
   71 instance Has Service.Passwd.Passwd ActionContext where
   72   view = (view . contextService)
   73 instance Has Service.Entropy.Entropy ActionContext where
   74   view = (view . contextService)
   75 instance Has Secret ActionContext where
   76   view = (view . contextService)
   77 instance Has InternalState ActionContext where
   78   view = contextResourceState
   79 instance Has DBConn ActionContext where
   80   view = contextDB

   82 type ActionContextM a = ReaderT ActionContext IO a
   83 
   84 -- | Perform an atomic action without an identity with a guaranteed database
   85 -- connection and a fixed version of 'now'.
   86 runContextM
   87     :: ActionContextM a
   88     -> Service
   89     -> IO a
   90 runContextM action rc = do
   91     t <- getCurrentTime
   92     runResourceT $ withInternalState $ \is ->
   93         withDB (serviceDB rc) $ runReaderT action . ActionContext rc t is

   95 -- | A ActionContext with no Identity.
   96 newtype BackgroundContext = BackgroundContext { backgroundContext :: ActionContext }
   97     deriving
   98         ( Has Service
   99         , Has Notifications
  100         , Has Solr
  101         , Has Ingest
  102         , Has HTTPClient
  103         , Has Storage
  104         , Has Logs
  105         , Has DBConn
  106         )
  107 
  108 instance Has Timestamp BackgroundContext where
  109   view = (contextTimestamp . backgroundContext)
  110 instance Has Identity BackgroundContext where
  111   view _ = IdentityNotNeeded
  112 instance Has SiteAuth BackgroundContext where
  113   view _ = view IdentityNotNeeded
  114 instance Has Party BackgroundContext where
  115   view _ = view IdentityNotNeeded
  116 instance Has (Id Party) BackgroundContext where
  117   view _ = view IdentityNotNeeded
  118 instance Has Access BackgroundContext where
  119   view _ = view IdentityNotNeeded
  120 
  121 type BackgroundContextM a = ReaderT BackgroundContext IO a
  122 
  123 withBackgroundContextM :: BackgroundContextM a -> ActionContextM a
  124 withBackgroundContextM = withReaderT BackgroundContext

  126 -- | A ActionContext with no Identity, for running Solr indexing.
  127 data SolrIndexingContext = SolrIndexingContext
  128   { slcLogs :: !Logs
  129   , slcHTTPClient :: !HTTPClient
  130   , slcSolr :: !Solr
  131   , slcDB :: !DBConn -- ^ The specific connection chosen for the running action?
  132   }
  133 
  134 instance Has Solr SolrIndexingContext where
  135   view = slcSolr
  136 instance Has Logs SolrIndexingContext where
  137   view = slcLogs
  138 instance Has HTTPClient SolrIndexingContext where
  139   view = slcHTTPClient
  140 instance Has DBConn SolrIndexingContext where
  141   view = slcDB
  142 
  143 instance Has Identity SolrIndexingContext where
  144   view _ = IdentityNotNeeded
  145 instance Has SiteAuth SolrIndexingContext where
  146   view _ = view IdentityNotNeeded
  147 instance Has Party SolrIndexingContext where
  148   view _ = view IdentityNotNeeded
  149 instance Has (Id Party) SolrIndexingContext where
  150   view _ = view IdentityNotNeeded
  151 instance Has Access SolrIndexingContext where
  152   view _ = view IdentityNotNeeded
  153 
  154 type SolrIndexingContextM a = ReaderT SolrIndexingContext IO a
  155 
  156 -- | Build a simpler SolrIndexingContext from a complete ActionContext
  157 mkSolrIndexingContext :: ActionContext -> SolrIndexingContext
  158 mkSolrIndexingContext ac =
  159     SolrIndexingContext {
  160           slcLogs = (serviceLogs . contextService) ac
  161         , slcHTTPClient = (serviceHTTPClient . contextService) ac
  162         , slcSolr = (serviceSolr . contextService) ac
  163         , slcDB = contextDB ac
  164     }
