  151 blankRecord :: Category -> Volume -> Record
  152 blankRecord cat vol = Record
  153   { recordRow = RecordRow
  154     { recordId = error "blankRecord"
  155     , recordCategory = cat
  156     }
  157   , recordVolume = vol
  158   , recordRelease = Nothing
  159   , recordMeasures = []
  160   }
