   38 blankComment :: Account -> Slot -> Comment
   39 blankComment who slot = Comment
   40   { commentId = error "blankComment"
   41   , commentWho = who
   42   , commentSlot = slot
   43   , commentTime = error "blankComment"
   44   , commentText = ""
   45   , commentParents = []

   48 lookupComment :: (MonadDB c m, MonadHasIdentity c m) => Id Comment -> m (Maybe Comment)
   49 lookupComment i = do
   50   ident <- peek
   51   dbQuery1 $(selectQuery (selectComment 'ident) "$!WHERE comment.id = ${i}")

   53 lookupSlotComments :: (MonadDB c m, MonadHasIdentity c m) => Slot -> Int -> m [Comment]
   54 lookupSlotComments (Slot c s) n = do
   55   ident <- peek
   56   dbQuery $ ($ c) <$> $(selectQuery (selectContainerComment 'ident) "$!WHERE comment.container = ${containerId $ containerRow c} AND comment.segment && ${s} ORDER BY comment.thread LIMIT ${fromIntegral n :: Int64}")

   58 mapQuery :: ByteString -> ([PGValue] -> a) -> PGSimpleQuery a
   59 mapQuery qry mkResult =
   60   fmap mkResult (rawPGSimpleQuery qry)

  188 commentJSON :: JSON.ToNestedObject o u => Comment -> JSON.Record (Id Comment) o
  189 commentJSON Comment{ commentSlot = Slot{..}, ..} = JSON.Record commentId $
  190      "container" JSON..=: containerJSON False slotContainer -- should compute based on volume
  191   <> segmentJSON slotSegment
  192   <> "who" JSON..=: partyJSON (accountParty commentWho)
  193   <> "time" JSON..= commentTime
  194   <> "text" JSON..= commentText
  195   <> "parents" `JSON.kvObjectOrEmpty` (if null commentParents then empty else pure commentParents)
