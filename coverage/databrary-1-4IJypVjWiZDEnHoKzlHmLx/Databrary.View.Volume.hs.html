   33 htmlVolumeViewLink :: QueryLike q => VolumeRow -> q -> H.Html
   34 htmlVolumeViewLink v q =
   35   H.a H.! actionLink viewVolume (HTML, volumeId v) q
   36     $ H.text $ volumeName v
   37 
   38 htmlVolumeView :: Volume -> [Tag] -> RequestContext -> H.Html
   39 htmlVolumeView v t req = htmlTemplate req Nothing $ \js -> do
   40   H.div H.! H.customAttribute "typeof" "dataset" $ do
   41     H.h1 H.! H.customAttribute "property" "name" $ H.text $ volumeName $ volumeRow v
   42     when (view v >= PermissionEDIT) $
   43       H.p $
   44         H.a H.! actionLink viewVolumeEdit (volumeId $ volumeRow v) js $ "edit"
   45     H.img
   46       H.! HA.src (builderValue $ actionURL Nothing thumbVolume (volumeId $ volumeRow v) [])
   47     H.dl $ do
   48       forM_ (getVolumeAlias v) $ \a -> do
   49         H.dt "alias"
   50         H.dd H.! H.customAttribute "property" "alternateName" $ H.text a
   51       forM_ (volumeOwners v) $ \(p, n) -> do
   52         H.dt "owner"
   53         H.dd H.! H.customAttribute "property" "creator" $ H.a H.! actionLink viewParty (HTML, TargetParty p) js $ H.text n
   54       forM_ (volumeBody $ volumeRow v) $ \b -> do
   55         H.dt "body"
   56         H.dd H.! H.customAttribute "property" "description" $ H.text b -- format
   57       forM_ (volumeDOI $ volumeRow v) $ \d -> do
   58         H.dt "doi"
   59         H.dd H.! H.customAttribute "property" "alternateName" $ byteStringHtml d
   60       H.dt "keywords"
   61       H.dd $ H.ul H.! HA.class_ "comma" H.! H.customAttribute "property" "keywords" $ do
   62         forM_ t $ \n  -> do
   63           H.li $ byteStringHtml $ tagNameBS $ tagName n

   65 htmlVolumeForm :: Maybe Volume -> Maybe Citation -> FormHtml f
   66 htmlVolumeForm vol cite = do
   67   field "name" $ inputText $ volumeName . volumeRow <$> vol
   68   field "alias" $ inputText $ volumeAlias . volumeRow =<< vol
   69   field "body" $ inputTextarea $ volumeBody . volumeRow =<< vol
   70   "citation" .:> do
   71     field "head" $ inputText $ citationHead <$> cite
   72     field "url" $ inputText $ fmap show $ citationURL =<< cite
   73     field "year" $ inputText $ fmap show $ citationYear =<< cite
   74 
   75 htmlVolumeEdit :: Maybe (Volume, Maybe Citation) -> RequestContext -> FormHtml f
   76 htmlVolumeEdit Nothing = htmlForm "Create volume" createVolume HTML (htmlVolumeForm Nothing Nothing) (const mempty)
   77 htmlVolumeEdit (Just (v, cite)) = htmlForm ("Edit " <> volumeName (volumeRow v)) postVolume (HTML, volumeId $ volumeRow v) (htmlVolumeForm (Just v) cite) (const mempty)
   78 
   79 htmlVolumeLinksEdit :: Volume -> [Citation] -> RequestContext -> FormHtml f
   80 htmlVolumeLinksEdit vol links = htmlForm "Edit volume links" postVolumeLinks (HTML, volumeId $ volumeRow vol)
   81   (withSubFormsViews links $ \link -> do
   82     field "head" $ inputText $ citationHead <$> link
   83     field "url" $ inputText $ fmap show $ citationURL =<< link)
   84   (const mempty)

   86 htmlVolumeList :: JSOpt -> [Volume] -> H.Html
   87 htmlVolumeList js vl = H.ul
   88   H.! HA.class_ "flat"
   89   $ forM_ vl $ \v -> H.li
   90     $ H.article
   91       H.! HA.class_ "volume-list-result cf"
   92       $ do
   93         H.h1
   94           $ htmlVolumeViewLink (volumeRow v) js
   95         H.ul
   96           H.! HA.class_ "flat semicolon"
   97           $ forM_ (volumeOwners v) $ \(p, o) -> H.li $ do
   98             H.a H.! actionLink viewParty (HTML, TargetParty p) js
   99               $ H.text o
  100         mapM_ (H.p . H.text) $ volumeBody $ volumeRow v

  102 htmlVolumeSearch :: VolumeFilter -> [Volume] -> RequestContext -> FormHtml f
  103 htmlVolumeSearch VolumeFilter{..} vl req = htmlForm "Volume search" queryVolumes HTML
  104   (field "query" $ inputText volumeFilterQuery)
  105   (\js -> htmlPaginate (htmlVolumeList js) volumeFilterPaginate vl (view req))
  106   req
