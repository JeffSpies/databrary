   56 confPGDatabase :: C.Config -> PGDatabase
   57 confPGDatabase conf = defaultPGDatabase
   58   { pgDBHost = fromMaybe "localhost" host
   59   , pgDBPort = if isJust host
   60      then PortNumber (maybe 5432 fromInteger $ conf C.! "port")
   61      else UnixSocket (fromJust $ conf C.! "sock")
   62   , pgDBName = fromMaybe user $ conf C.! "db"
   63   , pgDBUser = user
   64   , pgDBPass = fromMaybe "" $ conf C.! "pass"
   65   , pgDBDebug = fromMaybe False $ conf C.! "debug"
   66   }
   67   where
   68   host = conf C.! "host"
   69   user = conf C.! "user"

   72 data DBPool = DBPool (Pool PGConnection) (Pool PGSimple.Connection)
   73 type DBConn = PGConnection
   74 
   75 initDB :: C.Config -> IO DBPool
   76 initDB conf =
   77     DBPool
   78         <$> (createPool' (pgConnect db) pgDisconnect)
   79         <*> (createPool' (PGSimple.connect simpleConnInfo) (PGSimple.close))
   80   where
   81     createPool' :: IO a -> (a -> IO ()) -> IO (Pool a)
   82     createPool' get release =
   83         createPool get release stripes (fromInteger idle) conn
   84     db = confPGDatabase conf
   85     simpleConnInfo = PGSimple.defaultConnectInfo
   86         { PGSimple.connectHost     = pgDBHost db
   87         , PGSimple.connectPort     = case pgDBPort db of
   88             PortNumber x -> fromIntegral x -- x is opaque
   89             UnixSocket _ ->
   90                 PGSimple.connectPort PGSimple.defaultConnectInfo
   91             Service _ ->
   92                 PGSimple.connectPort PGSimple.defaultConnectInfo
   93         , PGSimple.connectUser     = BS.unpack (pgDBUser db)
   94         , PGSimple.connectPassword = BS.unpack (pgDBPass db)
   95         , PGSimple.connectDatabase = BS.unpack (pgDBName db)
   96         }
   97     stripes = fromMaybe 1 $ conf C.! "stripes"
   98     idle    = fromMaybe 300 $ conf C.! "idle"
   99     conn    = fromMaybe 16 $ conf C.! "maxconn"

  101 finiDB :: DBPool -> IO ()
  102 finiDB (DBPool p p') = do
  103    -- Different types -> no 'travers'ing
  104    destroyAllResources p
  105    destroyAllResources p'

  108 withDB (DBPool p _) = withResource p

  116 type DBM a = ReaderT PGConnection IO a

  118 runDBM :: DBPool -> DBM a -> IO a
  119 runDBM p = withDB p . runReaderT

  121 liftDBM :: MonadDB c m => DBM a -> m a
  122 liftDBM = liftDB . runReaderT

  124 -- |Combination of 'liftDBM' and lifted 'tryJust'
  125 dbTryJust :: MonadDB c m => (PGError -> Maybe e) -> DBM a -> m (Either e a)
  126 dbTryJust err q = liftDB $ tryJust err . runReaderT q

  128 dbRunQuery :: (MonadDB c m, PGQuery q a) => q -> m (Int, [a])
  129 dbRunQuery q = liftDB $ \c -> pgRunQuery c q

  131 dbExecute :: (MonadDB c m, PGQuery q ()) => q -> m Int
  132 dbExecute q = liftDB $ \c -> pgExecute c q

  134 dbExecuteSimple :: MonadDB c m => PGSimpleQuery () -> m Int
  135 dbExecuteSimple = dbExecute

  137 dbExecute1 :: (MonadDB c m, PGQuery q (), Show q) => q -> m Bool
  138 dbExecute1 q = do
  139   r <- dbExecute q
  140   case r of
  141     0 -> return False
  142     1 -> return True
  143     _ -> fail $ "pgExecute1 " ++ show q ++ ": " ++ show r ++ " rows"
  144 
  145 dbExecute1' :: (MonadDB c m, PGQuery q (), Show q) => q -> m ()
  146 dbExecute1' q = do
  147   r <- dbExecute1 q
  148   unless r $ fail $ "pgExecute1' " ++ show q ++ ": failed"
  149 
  150 dbExecute_ :: (MonadDB c m) => BSL.ByteString -> m ()
  151 dbExecute_ q = liftDB $ \c -> pgSimpleQueries_ c q
  152 
  153 dbQuery :: (MonadDB c m, PGQuery q a) => q -> m [a]
  154 dbQuery q = liftDB $ \c -> pgQuery c q
  155 
  156 dbQuery1 :: (MonadDB c m, PGQuery q a, Show q) => q -> m (Maybe a)
  157 dbQuery1 q = do
  158   r <- dbQuery q
  159   case r of
  160     [] -> return $ Nothing
  161     [x] -> return $ Just x
  162     _ -> fail $ "pgQuery1 " ++ show q ++ ": too many results"
  163 
  164 dbQuery1' :: (MonadDB c m, PGQuery q a, Show q) => q -> m a
  165 dbQuery1' q = maybe (fail $ "pgQuery1' " ++ show q ++ ": no results") return =<< dbQuery1 q
  166 
  167 dbTransaction :: MonadDB c m => DBM a -> m a
  168 dbTransaction f = liftDB $ \c -> pgTransaction c (runReaderT f c)
  169 
  170 dbTransaction' :: (MonadBaseControl IO m, MonadDB c m) => m a -> m a
  171 dbTransaction' f = do
  172   c <- peek
  173   liftBaseOp_ (pgTransaction c) f

  177 loadPGDatabase :: IO PGDatabase
  178 loadPGDatabase = confPGDatabase . C.get "db" <$> C.load "databrary.conf"
  179 
  180 runDBConnection :: DBM a -> IO a
  181 runDBConnection f = bracket
  182   (pgConnect =<< loadPGDatabase)
  183   pgDisconnect
  184   (runReaderT f)
  185 
  186 loadTDB :: TH.DecsQ
  187 loadTDB = do
  188   database <- TH.runIO loadPGDatabase
  189   useTPGDatabase database
  190 
  191 {-# NOINLINE usedTDB #-}
  192 usedTDB :: IORef Bool
  193 usedTDB = unsafePerformIO $ newIORef False
  194 useTDB :: TH.DecsQ
  195 useTDB = do
  196   d <- TH.runIO $ atomicModifyIORef' usedTDB ((,) True)
  197   if d
  198     then return []
  199     else loadTDB
  200 
  201 runTDB :: DBM a -> TH.Q a
  202 runTDB f = do
  203   _ <- useTDB
  204   TH.runIO $ withTPGConnection $ runReaderT f
