   24 releaseTitle :: Maybe Release -> Messages -> T.Text
   25 releaseTitle rel = getMessage $ C.Path ["release", maybe "UNRELEASED" (BSC.pack . show) rel, "title"]
   26 
   27 htmlContainerForm :: Maybe Container -> FormHtml f
   28 htmlContainerForm cont = do
   29   field "name" $ inputText (containerName . containerRow =<< cont)
   30   field "date" $ inputDate (containerDate . containerRow =<< cont)
   31   field "release" $ inputEnum False (containerRelease =<< cont)
   32 
   33 htmlContainerEdit :: Either Volume Container -> RequestContext -> FormHtml f
   34 htmlContainerEdit (Left v)  = htmlForm "Create container" createContainer (HTML, volumeId $ volumeRow v) (htmlContainerForm Nothing) (const mempty)
   35 htmlContainerEdit (Right c) = htmlForm ("Edit container " <> fold (containerName $ containerRow c)) postContainer (HTML, containerSlotId $ containerId $ containerRow c) (htmlContainerForm $ Just c) (const mempty)
