   33 slotAssetRow :: Selector -- ^ @'Segment'@
   34 slotAssetRow = selectColumn "slot_asset" "segment"

   36 makeSlotAsset :: Asset -> Container -> Segment -> AssetSlot
   37 makeSlotAsset a c s = AssetSlot a (Just (Slot c s))
   38 
   39 _selectAssetContainerSlotAsset :: Selector -- ^ @'Asset' -> 'Container' -> 'AssetSlot'@
   40 _selectAssetContainerSlotAsset = selectMap (TH.VarE 'makeSlotAsset `TH.AppE`) slotAssetRow
   41 
   42 makeContainerSlotAsset :: Segment -> AssetRow -> Container -> AssetSlot
   43 makeContainerSlotAsset s ar c = makeSlotAsset (Asset ar $ view c) c s
   44 
   45 selectContainerSlotAsset :: Selector -- ^ @'Container' -> 'AssetSlot'@
   46 selectContainerSlotAsset = selectJoin 'makeContainerSlotAsset
   47   [ slotAssetRow
   48   , joinOn "slot_asset.asset = asset.id" selectAssetRow -- XXX volumes match?
   49   ]

   57 makeVolumeSlotIdAsset :: SlotId -> AssetRow -> Volume -> (Asset, SlotId)
   58 makeVolumeSlotIdAsset s ar v = (Asset ar v, s)
   59 
   60 selectVolumeSlotIdAsset :: Selector -- ^ @'Volume' -> ('Asset', 'SlotId')@
   61 selectVolumeSlotIdAsset = selectJoin 'makeVolumeSlotIdAsset
   62   [ selectColumns 'SlotId "slot_asset" ["container", "segment"]
   63   , joinOn "slot_asset.asset = asset.id"
   64     selectAssetRow -- XXX volumes match?

   67 makeAssetSlotAsset :: Segment -> (Volume -> Container) -> Asset -> AssetSlot
   68 makeAssetSlotAsset s cf a = makeSlotAsset a (cf (view a)) s

   70 selectAssetSlotAsset :: Selector -- ^ @'Asset' -> 'AssetSlot'@
   71 selectAssetSlotAsset = selectJoin 'makeAssetSlotAsset
   72   [ slotAssetRow
   73   , joinOn "slot_asset.container = container.id"
   74     selectVolumeContainer -- XXX volumes match?
   75   ]

   77 makeVolumeSlotAsset :: Segment -> AssetRow -> (Volume -> Container) -> Volume -> AssetSlot
   78 makeVolumeSlotAsset s ar cf v = makeSlotAsset (Asset ar v) (cf v) s

   80 selectVolumeSlotAsset :: Selector -- ^ @'Volume' -> 'AssetSlot'@
   81 selectVolumeSlotAsset = selectJoin 'makeVolumeSlotAsset
   82   [ slotAssetRow
   83   , joinOn "slot_asset.asset = asset.id"
   84     selectAssetRow
   85   , joinOn "slot_asset.container = container.id AND asset.volume = container.volume"
   86     selectVolumeContainer

   89 selectSlotAsset :: TH.Name -- ^ @'Identity'@
   90   -> Selector -- ^ @'AssetSlot'@
   91 selectSlotAsset ident = selectJoin '($)
   92   [ selectVolumeSlotAsset
   93   , joinOn "asset.volume = volume.id"
   94     $ selectVolume ident
   95   ]

   97 makeVolumeAssetSlot :: AssetRow -> Maybe (Asset -> AssetSlot) -> Volume -> AssetSlot
   98 makeVolumeAssetSlot ar sf = fromMaybe assetNoSlot sf . Asset ar

  100 selectVolumeAssetSlot :: Selector -- ^ @'Volume' -> 'AssetSlot'@
  101 selectVolumeAssetSlot = selectJoin 'makeVolumeAssetSlot

  103   , maybeJoinOn "asset.id = slot_asset.asset AND asset.volume = container.volume"

  107 selectAssetSlot :: TH.Name -- ^ @'Identity'@
  108   -> Selector -- ^ @'AssetSlot'@

  115 slotAssetKeys :: String -- ^ @'AssetSlot'@
  116   -> [(String, String)]
  117 slotAssetKeys as =
  118   [ ("asset", "${assetId $ assetRow $ slotAsset " ++ as ++ "}") ]
  119 
  120 slotAssetSets :: String -- ^ @'AssetSlot'@
  121   -> [(String, String)]
  122 slotAssetSets as =
  123   [ ("container", "${containerId . containerRow . slotContainer <$> assetSlot " ++ as ++ "}")
  124   , ("segment", "${slotSegment <$> assetSlot " ++ as ++ "}")
  125   ]

  127 insertSlotAsset :: TH.Name -- ^ @'AuditIdentity'@
  128   -> TH.Name -- ^ @'AssetSlot'@
  129   -> TH.ExpQ
  130 insertSlotAsset ident o = auditInsert ident "slot_asset"
  131   (slotAssetKeys os ++ slotAssetSets os)
  132   Nothing
  133   where os = nameRef o
  134 
  135 updateSlotAsset :: TH.Name -- ^ @'AuditIdentity'@
  136   -> TH.Name -- ^ @'AssetSlot'@
  137   -> TH.ExpQ
  138 updateSlotAsset ident o = auditUpdate ident "slot_asset"
  139   (slotAssetSets os)
  140   (whereEq $ slotAssetKeys os)
  141   Nothing
  142   where os = nameRef o
  143 
  144 deleteSlotAsset :: TH.Name -- ^ @'AuditIdentity'@
  145   -> TH.Name -- ^ @'AssetSlot'@
  146   -> TH.ExpQ
  147 deleteSlotAsset ident o = auditDelete ident "slot_asset"
  148   (whereEq $ slotAssetKeys os)
  149   Nothing
  150   where os = nameRef o
