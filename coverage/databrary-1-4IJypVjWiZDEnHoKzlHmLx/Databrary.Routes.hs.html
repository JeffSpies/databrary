  195     hn0 :: Action -> WAR.Handler IO  -- make handler
  196     hn0 act = \_ req responder -> runAction routeContext act req responder
  197     hn :: ([(BS.ByteString, BS.ByteString)] -> Action) -> WAR.Handler IO  -- make handler
  198     hn mkAction = \ps req responder -> runAction routeContext (mkAction ps) req responder
  199     hnm :: (HTM.Method -> [(BS.ByteString, BS.ByteString)] -> Action) -> WAR.Handler IO  -- make handler with method
  200     hnm mkAction = \ps req responder -> runAction routeContext (mkAction (WAI.requestMethod req) ps) req responder
