   17 htmlSiteActivity :: SiteStats -> RequestContext -> H.Html
   18 htmlSiteActivity SiteStats{..} req = htmlTemplate req (Just "activity") $ \_ -> do
   19   H.ul $ do
   20     H.li $ sh (statsAuthorizedSite A.! PermissionEDIT) >> " authorized PIs"
   21     H.li $ sh (statsAuthorizedSite A.! PermissionREAD) >> " affiliates who have access to the shared data"
   22     H.li $ sh (statsAuthorizedSite A.! PermissionADMIN) >> " universities"
   23     H.li $ sh statsVolumes >> " volumes, " >> sh statsVolumesShared >> " have been opened for sharing"
   24     H.li $ sh statsAssetDuration >> " of recordings and " >> sh statsAssets >> " files from " >> sh (maybe 0 ((statsRecords M.!) . categoryId) participant) >> " participants"
   25   return ()
   26   where
   27   participant = find (("participant" ==) . categoryName) allCategories
   28   sh :: Show a => a -> H.Markup
   29   sh = H.string . show
