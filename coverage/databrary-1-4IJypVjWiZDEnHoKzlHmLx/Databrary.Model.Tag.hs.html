   95 lookupVolumeTagUseRows :: MonadDB c m => Volume -> m [TagUseRow]
   96 lookupVolumeTagUseRows v =
   97   dbQuery $(selectQuery selectTagUseRow "JOIN container ON tag_use.container = container.id WHERE container.volume = ${volumeId $ volumeRow v} ORDER BY container.id")

   99 addTagUse :: MonadDB c m => TagUse -> m Bool
  100 addTagUse t = either (const False) id <$> do
  101   dbTryJust (guard . isExclusionViolation)
  102     $ dbExecute1 (if tagKeyword t
  103       then $(insertTagUse True 't)
  104       else $(insertTagUse False 't))
  105 
  106 removeTagUse :: MonadDB c m => TagUse -> m Int
  107 removeTagUse t =
  108   dbExecute
  109     (if tagKeyword t
  110       then $(deleteTagUse True 't)
  111       else $(deleteTagUse False 't))

  113 lookupTopTagWeight :: MonadDB c m => Int -> m [TagWeight]
  114 lookupTopTagWeight lim =
  115   dbQuery $(selectQuery (selectTagWeight "") "$!ORDER BY weight DESC LIMIT ${fromIntegral lim :: Int64}")

  117 emptyTagCoverage :: Tag -> Container -> TagCoverage
  118 emptyTagCoverage t c = TagCoverage (TagWeight t 0) c [] [] []

  120 lookupTagCoverage :: (MonadDB c m, MonadHasIdentity c m) => Tag -> Slot -> m TagCoverage
  121 lookupTagCoverage t (Slot c s) = do
  122   ident <- peek
  123   fromMaybe (emptyTagCoverage t c) <$> dbQuery1 (($ c) . ($ t) <$> $(selectQuery (selectTagCoverage 'ident "WHERE container = ${containerId $ containerRow c} AND segment && ${s} AND tag = ${tagId t}") "$!"))
  124 
  125 lookupSlotTagCoverage :: (MonadDB c m, MonadHasIdentity c m) => Slot -> Int -> m [TagCoverage]
  126 lookupSlotTagCoverage slot lim = do
  127   ident <- peek
  128   dbQuery $(selectQuery (selectSlotTagCoverage 'ident 'slot) "$!ORDER BY weight DESC LIMIT ${fromIntegral lim :: Int64}")

  130 lookupSlotKeywords :: (MonadDB c m) => Slot -> m [Tag]
  131 lookupSlotKeywords Slot{..} =
  132   dbQuery $(selectQuery selectTag "JOIN keyword_use ON id = tag WHERE container = ${containerId $ containerRow slotContainer} AND segment = ${slotSegment}")

  138 tagCoverageJSON :: JSON.ToObject o => TagCoverage -> JSON.Record TagName o
  139 tagCoverageJSON TagCoverage{..} = tagWeightJSON tagCoverageWeight `JSON.foldObjectIntoRec`
