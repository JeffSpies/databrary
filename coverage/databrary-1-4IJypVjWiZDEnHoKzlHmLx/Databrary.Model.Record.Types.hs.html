  138 blankRecord :: Category -> Volume -> Record
  139 blankRecord cat vol = Record
  140   { recordRow = RecordRow
  141     { recordId = error "blankRecord"
  142     , recordCategory = cat
  143     }
  144   , recordVolume = vol
  145   , recordRelease = Nothing
  146   , recordMeasures = []
  147   }
