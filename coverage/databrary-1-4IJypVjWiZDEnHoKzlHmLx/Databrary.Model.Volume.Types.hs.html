   25 type instance IdType Volume = Int32

   27 data VolumeRow = VolumeRow
   28   { volumeId :: Id Volume
   29   , volumeName :: T.Text
   30   , volumeBody :: Maybe T.Text
   31   , volumeAlias :: Maybe T.Text
   32   , volumeDOI :: Maybe BS.ByteString
   33   }
   34   deriving (Show, Eq)
   35 
   36 type VolumeOwner = (Id Party, T.Text)
   37 
   38 data Volume = Volume
   39   { volumeRow :: !VolumeRow
   40   , volumeCreation :: Timestamp
   41   , volumeOwners :: [VolumeOwner]
   42   , volumePermission :: Permission
   43   , volumeAccessPolicy :: VolumeAccessPolicy
   44   }
   45   deriving (Show, Eq)
   46 
   47 instance Kinded Volume where
   48   kindOf _ = "volume"
   49

   62 volumePermissionPolicy :: Volume -> (Permission, VolumeAccessPolicy)
   63 volumePermissionPolicy Volume{..} =
   64   ( volumePermission
   65   , volumeAccessPolicy )
   66 
   67 volumeAccessPolicyWithDefault :: Permission -> Maybe Bool -> VolumeAccessPolicy
   68 volumeAccessPolicyWithDefault perm1 mShareFull =
   69   case perm1 of
   70     PermissionPUBLIC ->
   71       let shareFull = fromMaybe True mShareFull -- assume true because historically volumes were public full
   72       in if shareFull then PermLevelDefault else PublicRestricted
   73     _ ->
   74       PermLevelDefault
   75

   77 blankVolume :: Volume
   78 blankVolume = Volume
   79   { volumeRow = VolumeRow
   80     { volumeId = error "blankVolume"
   81     , volumeName = ""
   82     , volumeAlias = Nothing
   83     , volumeBody = Nothing
   84     , volumeDOI = Nothing
   85     }
   86   , volumeCreation = posixSecondsToUTCTime 1357900000
   87   , volumeOwners = []
   88   , volumePermission = PermissionNONE
   89   , volumeAccessPolicy = PermLevelDefault
   90   }
