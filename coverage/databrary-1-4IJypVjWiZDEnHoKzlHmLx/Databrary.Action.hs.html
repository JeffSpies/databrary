   45 redirectRouteResponse :: Status -> ResponseHeaders -> R.RouteAction r a -> r -> Request -> Response
   46 redirectRouteResponse s h r a req =
   47   emptyResponse s ((hLocation, BSL.toStrict $ BSB.toLazyByteString $ actionURL (Just req) r a (Wai.queryString req)) : h)

   49 otherRouteResponse :: ResponseHeaders -> R.RouteAction r a -> r -> Request -> Response

   76 runActionRoute
   77     :: R.RouteMap Action
   78     -> (Service -> [(BS.ByteString, WAR.Handler IO)])
   79     -> Service
   80     -> Wai.Application
   81 runActionRoute routeMap mkNewRouteMap routeContext req =
   82     let eMatchedAction :: Either (Status, ResponseHeaders) Action
   83         eMatchedAction = R.routeWai req routeMap
   84     in
   85       case eMatchedAction of
   86         Right act ->
   87             runAction routeContext act req
   88         Left (st,hdrs) ->
   89             if st == notFound404 -- currently, this might be only possible error result?
   90             then
   91                 api1App (mkNewRouteMap routeContext) req
   92             else
   93                 runAction routeContext (err (st,hdrs)) req
   94   where
   95     err :: (Status, ResponseHeaders) -> Action
   96     err (status, headers) = withoutAuth $ peeks $ response status headers . htmlNotFound
