   22 lookupSlot :: (MonadDB c m, MonadHasIdentity c m) => Id Slot -> m (Maybe Slot)
   23 lookupSlot (Id (SlotId c s)) =
   24   fmap (`Slot` s) <$> lookupContainer c
   25 
   26 auditSlotDownload :: MonadAudit c m => Bool -> Slot -> m ()
   27 auditSlotDownload success Slot{ slotContainer = c, slotSegment = seg } = do
   28   ai <- getAuditIdentity
   29   dbExecute1' [pgSQL|$INSERT INTO audit.slot (audit_action, audit_user, audit_ip, container, segment) VALUES
   30     (${if success then AuditActionOpen else AuditActionAttempt}, ${auditWho ai}, ${auditIp ai}, ${containerId $ containerRow c}, ${seg})|]
   31 
   32 slotJSON :: JSON.ToObject o => Slot -> JSON.Record (Id Container) o
   33 slotJSON Slot{..} = containerJSON False slotContainer -- probably add bool to slotJSON
   34   `JSON.foldObjectIntoRec` (segmentJSON slotSegment)
