   18 htmlRecordForm :: Volume -> RequestContext -> FormHtml f
   19 htmlRecordForm vol = htmlForm "Create record"
   20   createRecord (HTML, volumeId $ volumeRow vol)
   21   (field "category" $ inputSelect Nothing $ ("", "<record>") : map (\c -> (BSC.pack $ show $ categoryId c, categoryName c)) allCategories)
   22   (const mempty)
   23 
   24 htmlRecordMeasureForm :: Record -> Metric -> RequestContext -> FormHtml f
   25 htmlRecordMeasureForm rec met = htmlForm "Set measure"
   26   postRecordMeasure (HTML, recordId $ recordRow rec, metricId met)
   27   (field "datum" $ inputText (Nothing :: Maybe String))
   28   (const mempty)
