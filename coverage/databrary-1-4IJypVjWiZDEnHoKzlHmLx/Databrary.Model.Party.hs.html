  100 partyRowJSON :: JSON.ToObject o => PartyRow -> JSON.Record (Id Party) o
  101 partyRowJSON PartyRow{..} = JSON.Record partyId $
  102      "sortname" JSON..= partySortName
  103   <> "prename" `JSON.kvObjectOrEmpty` partyPreName
  104   <> "orcid" `JSON.kvObjectOrEmpty` (show <$> partyORCID)
  105   <> "affiliation" `JSON.kvObjectOrEmpty` partyAffiliation
  106   <> "url" `JSON.kvObjectOrEmpty` partyURL
  107 
  108 partyJSON :: JSON.ToObject o => Party -> JSON.Record (Id Party) o
  109 partyJSON p@Party{..} = partyRowJSON partyRow `JSON.foldObjectIntoRec`
  110  (   "institution" `JSON.kvObjectOrEmpty` (True `useWhen` (isNothing partyAccount))
  111   <> "email" `JSON.kvObjectOrEmpty` partyEmail p
  112   <> "permission" `JSON.kvObjectOrEmpty` (partyPermission `useWhen` (partyPermission > PermissionREAD)))

  941 data PartyFilter = PartyFilter
  942   { partyFilterQuery :: Maybe String
  943   , partyFilterAuthorization :: Maybe Permission
  944   , partyFilterInstitution :: Maybe Bool
  945   , partyFilterPaginate :: Paginate
  946   }
  947 
  948 instance Monoid PartyFilter where
  949   mempty = PartyFilter Nothing Nothing Nothing def
  950   mappend (PartyFilter q1 a1 i1 p) (PartyFilter q2 a2 i2 _) =
  951     PartyFilter (q1 <> q2) (a1 <|> a2) (i1 <|> i2) p

  953 partyFilter :: PartyFilter -> Identity -> BS.ByteString
  954 partyFilter PartyFilter{..} ident = BS.concat
  955   [ withq partyFilterAuthorization (const " JOIN authorize_view ON party.id = child AND parent = 0")
