   99 partyRowJSON :: JSON.ToObject o => PartyRow -> JSON.Record (Id Party) o
  100 partyRowJSON PartyRow{..} = JSON.Record partyId $
  101      "sortname" JSON..= partySortName
  102   <> "prename" `JSON.kvObjectOrEmpty` partyPreName
  103   <> "orcid" `JSON.kvObjectOrEmpty` (show <$> partyORCID)
  104   <> "affiliation" `JSON.kvObjectOrEmpty` partyAffiliation
  105   <> "url" `JSON.kvObjectOrEmpty` partyURL

  107 partyJSON :: JSON.ToObject o => Party -> JSON.Record (Id Party) o
  108 partyJSON p@Party{..} = partyRowJSON partyRow `JSON.foldObjectIntoRec`
  109  (   "institution" `JSON.kvObjectOrEmpty` (True `useWhen` (isNothing partyAccount))
  110   <> "email" `JSON.kvObjectOrEmpty` partyEmail p
  111   <> "permission" `JSON.kvObjectOrEmpty` (partyPermission `useWhen` (partyPermission > PermissionREAD)))

  225 -- | Create a new party without an account, intended for creating institution parties.
  226 addParty :: MonadAudit c m => Party -> m Party
  227 addParty bp = do
  228   ident <- getAuditIdentity
  229   -- Similar to add account, load resulting party with default values for party permission and
  230   -- access. 
  231   dbQuery1' $ fmap (\p -> Party p Nothing PermissionREAD Nothing) $(insertParty 'ident 'bp)

  233 -- | Create a new account without any authorizations, during registration, using the nobodySiteAuth.
  234 -- The account password will be blank. The party will not have any authorizations yet.
  235 addAccount :: MonadAudit c m => Account -> m Account
  236 addAccount ba@Account{ accountParty = bp } = do
  237   ident <- getAuditIdentity
  238   -- Create a party. The account will be created below, so start with no account.
  239   -- Load resulting party with default values for party permission and access for now.
  240   p <- dbQuery1' $ fmap (\p -> Party p Nothing PermissionREAD Nothing) $(insertParty 'ident 'bp)
  241   let pa = p{ partyAccount = Just a }
  242       a = ba{ accountParty = pa }
  243   -- Create an account with no password, and the email provided
  244   dbExecute1' $(insertAccount 'ident 'a)
  245   return a

  247 removeParty :: MonadAudit c m => Party -> m Bool
  248 removeParty p = do
  249   ident <- getAuditIdentity
  250   dbTransaction $ handleJust (guard . isForeignKeyViolation) (\_ -> return False) $ do
  251     _ <- dbExecute1 $(deleteAccount 'ident 'p)
  252     dbExecute1 $(deleteParty 'ident 'p)

  254 lookupFixedParty :: Id Party -> Identity -> Maybe Party
  255 lookupFixedParty (Id (-1)) _ = Just nobodyParty
  256 lookupFixedParty (Id 0) i =
  257   Just rootParty{
  258     partyPermission = accessPermission i `max` PermissionSHARED
  259   , partyAccess = (accessMember i > PermissionNONE) `thenUse` (view i) }
  260 lookupFixedParty i a = (view a) `useWhen` (i == view a)

  376 -- | resolve email to its party and enclosing account and site authenticated identity, possibly case insensitive
  377 lookupSiteAuthByEmail
  378     :: MonadDB c m
  379     => Bool -- ^ be case-insensitive?
  380     -> BS.ByteString
  381     -> m (Maybe SiteAuth)
  382 lookupSiteAuthByEmail caseInsensitive e = do
  383   r <- dbQuery1 $(selectQuery selectSiteAuth "WHERE account.email = ${e}")
  384   if caseInsensitive && isNothing r
  385     then do
  386       a <- dbQuery $(selectQuery selectSiteAuth "WHERE lower(account.email) = lower(${e}) LIMIT 2")
  387       return $ case a of
  388         [x] -> Just x
  389         _ -> Nothing
  390     else
  391       return r

  403 data PartyFilter = PartyFilter
  404   { partyFilterQuery :: Maybe String
  405   , partyFilterAuthorization :: Maybe Permission
  406   , partyFilterInstitution :: Maybe Bool
  407   , partyFilterPaginate :: Paginate
  408   }
  409 
  410 instance Monoid PartyFilter where
  411   mempty = PartyFilter Nothing Nothing Nothing def
  412   mappend (PartyFilter q1 a1 i1 p) (PartyFilter q2 a2 i2 _) =
  413     PartyFilter (q1 <> q2) (a1 <|> a2) (i1 <|> i2) p
  414 
  415 partyFilter :: PartyFilter -> Identity -> BS.ByteString
  416 partyFilter PartyFilter{..} ident = BS.concat
  417   [ withq partyFilterAuthorization (const " JOIN authorize_view ON party.id = child AND parent = 0")
  418   , " WHERE id > 0 AND id != ", pgLiteralRep (partyId $ partyRow $ staffParty)
  419   , withq partyFilterQuery (\n -> " AND " <> queryVal <> " ILIKE " <> pgLiteralRep (wordPat n))
  420   , withq partyFilterAuthorization (\a -> " AND site = " <> pgSafeLiteral a)
  421   , withq partyFilterInstitution (\i -> if i then " AND account.id IS NULL" else " AND account.password IS NOT NULL")
  422   , " ORDER BY name, prename "
  423   , paginateSQL partyFilterPaginate
  424   ]
  425   where
  426   withq v f = maybe "" f v
  427   wordPat = intercalate "%" . ("":) . (++[""]) . words
  428   queryVal
  429     | showEmail ident = "(COALESCE(prename || ' ', '') || name || COALESCE(' ' || email, ''))"
  430     | otherwise = "(COALESCE(prename || ' ', '') || name)"

  442 changeAvatar :: MonadAudit c m => Party -> Maybe Asset -> m Bool
  443 changeAvatar p Nothing = do
  444   ident <- getAuditIdentity
  445   dbExecute1 $(auditDelete 'ident "avatar" "party = ${partyId $ partyRow p}" Nothing)
  446 changeAvatar p (Just a) = do
  447   ident <- getAuditIdentity
  448   (0 <) . fst <$> updateOrInsert
  449     $(auditUpdate 'ident "avatar" [("asset", "${assetId $ assetRow a}")] "party = ${partyId $ partyRow p}" Nothing)
  450     $(auditInsert 'ident "avatar" [("asset", "${assetId $ assetRow a}"), ("party", "${partyId $ partyRow p}")] Nothing)
