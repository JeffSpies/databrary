   23 permissionVIEW :: Permission
   24 permissionVIEW = PermissionPUBLIC

   26 -- |Alias for READ. Grants full access to private data, bypassing consent permissions.
   27 permissionPRIVATE :: Permission
   28 permissionPRIVATE = PermissionREAD

   30 -- |The necessary permission level to read a data object with the given release.
   31 -- Equivalent to the SQL function read_permission.
   32 readPermission :: Release -> Permission
   33 readPermission ReleasePUBLIC   = PermissionPUBLIC
   34 readPermission ReleaseSHARED   = PermissionSHARED
   35 readPermission ReleaseEXCERPTS = PermissionSHARED
   36 readPermission ReleasePRIVATE  = permissionPRIVATE
   37 
   38 -- |The most restrictive data release level that the current user may access under the given permission.
   39 -- Equivalent to the SQL function read_release.  Inverse of 'readPermission' module meaning of @Nothing@.
   40 readRelease :: Permission -> Maybe Release
   41 readRelease PermissionNONE   = Nothing
   42 readRelease PermissionPUBLIC = Just ReleasePUBLIC
   43 readRelease PermissionSHARED = Just ReleaseSHARED
   44 readRelease _                = Just ReleasePRIVATE
   45 
   46 -- |The effective permission for data objects with the given attributes, effectively collapsing ineffective permissions NONE.
   47 releasePermission :: Release -> Permission -> Permission
   48 releasePermission effectiveReleaseOnData currentUserAllowedPermissionOnVolume
   49   | currentUserAllowedPermissionOnVolume >= readPermission effectiveReleaseOnData = currentUserAllowedPermissionOnVolume
   50   | otherwise = PermissionNONE

  106 dataPermission3 :: (a -> EffectiveRelease) -> (a -> (Permission, VolumeAccessPolicy)) -> a -> Permission
  107 dataPermission3 getObjEffectiveRelease getCurrentUserPermLevel obj =
  108   let
  109    effRelease = getObjEffectiveRelease obj
  110   in 
  111     case getCurrentUserPermLevel obj of
  112       (p@PermissionPUBLIC, PublicRestricted) -> releasePermission (effRelPrivate effRelease) p
  113       -- other levels that behave more like private (options: none, shared, read, edit, admin) ? 
  114       (p, _) -> releasePermission (effRelPublic effRelease) p
  115 
  116 canReadData :: (a -> EffectiveRelease) -> (a -> (Permission, VolumeAccessPolicy)) -> a -> Bool
  117 canReadData getObjEffectiveRelease getCurrentUserPermLevel obj =
  118   dataPermission3 getObjEffectiveRelease getCurrentUserPermLevel obj > PermissionNONE
