   23 makeComment :: Id Comment -> Segment -> Timestamp -> T.Text -> [Maybe (Id Comment)] -> Account -> Container -> Comment
   24 makeComment i s t x p w c = Comment i w (Slot c s) t x (map (fromMaybe (error "NULL comment thread")) p)
   25 
   26 commentRow :: Selector -- ^ @'Account' -> 'Container' -> 'Comment'@
   27 commentRow = selectColumns 'makeComment "comment" ["id", "segment", "time", "text", "thread"]
   28 
   29 selectAccountContainerComment :: Selector -- ^ @'Account' -> 'Container' -> 'Comment'@
   30 selectAccountContainerComment = fromMap ("comment_thread AS " ++) commentRow
   31 
   32 selectContainerComment :: TH.Name -- ^ @'Identity'@
   33   -> Selector -- ^ @'Container' -> 'Comment'@
   34 selectContainerComment ident = selectJoin '($)
   35   [ selectAccountContainerComment
   36   , joinOn "comment.who = account.id"
   37     $ selectAccount ident
   38   ]
   39 
   40 selectComment :: TH.Name -- ^ @'Identity'@
   41   -> Selector -- ^ @'Comment'@
   42 selectComment ident = selectJoin '($)
   43   [ selectContainerComment ident
   44   , joinOn "comment.container = container.id"
   45     $ selectContainer ident
   46   ]

   57   -> [(String, String)]
