   32 type PathParameter = Parameter PathString
   33 
   34 idIso :: IdType a I.<-> Id a
   35 idIso = [I.biCase|a <-> Id a|]
   36 
   37 pathIdWith :: forall a . (Kinded a) => PathParser (IdType a) -> PathParser (Id a)
   38 pathIdWith p = fromString (kindOf (undefined :: a)) >/> idIso >$< p
   39 
   40 pathId :: forall a . (PathParameter (IdType a), Kinded a) => PathParser (Id a)
   41 pathId = pathIdWith R.parameter
   42 
   43 -- | The target party for some action?
   44 data PartyTarget
   45   = TargetProfile -- ^ Actor's own party
   46   | TargetParty (Id Party) -- ^ Someone else's party
   47 
   48 
   49 pathPartyTarget :: R.Path PartyTarget
   50 pathPartyTarget = [I.biCase|
   51     Left () <-> TargetProfile
   52     Right i <-> TargetParty i
   53   |] >$< ("profile" |/| pathId)
   54 
   55 data AuthorizeTarget = AuthorizeTarget
   56   { authorizeApply :: Bool
   57   , authorizeTarget :: Id Party
   58   }
   59 
   60 pathAuthorizeTarget :: PathParser AuthorizeTarget
   61 pathAuthorizeTarget = [I.biCase|(a, t) <-> AuthorizeTarget a t|] >$<
   62   (I.isRight >$< ("authorize" |/| "apply")
   63    </> idIso >$< R.parameter)
   64 
   65 newtype VolumeAccessTarget = VolumeAccessTarget
   66   { volumeAccessTarget :: Id Party
   67   }
   68 
   69 pathVolumeAccessTarget :: PathParser VolumeAccessTarget
   70 pathVolumeAccessTarget = "access" >/> [I.biCase|i <-> VolumeAccessTarget (Id i)|] >$< R.parameter
   71 
   72 slotIdIso :: (Id Container, Segment) I.<-> SlotId
   73 slotIdIso = [I.biCase|(c, s) <-> SlotId c s|]
   74 
   75 pathSegment :: PathParser Segment
   76 pathSegment = fullSegment =/= R.parameter
   77 
   78 pathSlot :: PathParser SlotId
   79 pathSlot = slotIdIso >$< (idIso >$< R.parameter </> pathSegment)
   80 
   81 pathSlotId :: PathParser (Id Slot)
   82 pathSlotId = pathIdWith pathSlot
   83 
   84 data TagId = TagId
   85   { tagIdKeyword :: Bool
   86   , tagIdName :: TagName
   87   }
   88 
   89 pathTagId :: PathParser TagId
   90 pathTagId = [I.biCase|(b, t) <-> TagId b t|] >$<
   91   (I.isRight >$< ("tag" |/| "keyword") </> R.parameter)
