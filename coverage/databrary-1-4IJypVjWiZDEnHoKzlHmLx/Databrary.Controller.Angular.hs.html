   29 data JSOpt
   30   = JSDisabled
   31   | JSDefault
   32   | JSEnabled
   33   deriving (Eq, Ord)
   34 
   35 instance Default JSOpt where
   36   def = JSDefault
   37 
   38 instance Monoid JSOpt where
   39   mempty = JSDefault
   40   mappend JSDefault j = j
   41   mappend j _ = j
   42 
   43 instance QueryLike JSOpt where
   44   toQuery JSDisabled = [("js", Just "0")]
   45   toQuery JSDefault = []
   46   toQuery JSEnabled = [("js", Just "1")]
   47 
   48 jsEnable :: Bool -> JSOpt
   49 jsEnable False = JSDisabled
   50 jsEnable True = JSEnabled
   51 
   52 jsURL :: JSOpt -> Wai.Request -> (JSOpt, BSB.Builder)
   53 jsURL js req =
   54   second (encodePath' (Wai.pathInfo req) . (toQuery js ++))
   55   $ unjs $ Wai.queryString req where
   56   unjs [] = (JSDefault, [])
   57   unjs (("js",v):q) = (jsEnable (boolParameterValue v), snd $ unjs q)
   58   unjs (x:q) = second (x:) $ unjs q
   59 
   60 browserBlacklist :: Regex.Regex
   61 browserBlacklist = Regex.makeRegex
   62   ("^Mozilla/.* \\(.*\\<(MSIE [0-9]\\.[0-9]|AppleWebKit/.* Version/[0-5]\\..* Safari/)" :: String)
   63 
   64 angularEnable :: JSOpt -> Wai.Request -> Bool
   65 angularEnable JSDisabled = const False
   66 angularEnable JSDefault = not . any (Regex.matchTest browserBlacklist) . lookupRequestHeader hUserAgent
   67 angularEnable JSEnabled = const True
   68 
   69 angularRequest :: Wai.Request -> Maybe BSB.Builder
   70 angularRequest req = (angularEnable js req) `thenUse` nojs
   71   where (js, nojs) = jsURL JSDisabled req
   72 
   73 angularResult :: BS.ByteString -> BSB.Builder -> RequestContext -> IO ()
   74 angularResult version nojs auth = do
   75   debug <-
   76 #ifdef DEVEL
   77     boolQueryParameter "debug" (view auth) ?$> liftIO appWebJS
   78 #else
   79     return Nothing
   80 #endif
   81   cssDeps <- case debug of
   82     Just _ -> cssWebDeps True
   83     Nothing -> (:[]) <$> makeWebFilePath "all.min.css"
   84   jsDeps <- case debug of
   85     Just debug' -> do
   86       w <- webDeps True
   87       d <- makeWebFilePath "debug.js"
   88       return $ w ++ (d : debug')
   89     Nothing -> (:[]) <$> makeWebFilePath "all.min.js"
   90   result $ okResponse [] (htmlAngular version cssDeps jsDeps nojs auth)
   91 
   92 angular :: Handler ()
   93 angular = do
   94   (servWeb :: Web) <- peek
   95   let version = getWebVersion servWeb
   96   (b :: Maybe BSB.Builder) <- peeks angularRequest
   97   mapM_
   98     (\nojsBldr ->
   99        let
  100          angularResultNoJs :: RequestContext -> IO ()
  101          angularResultNoJs = angularResult version nojsBldr
  102        in
  103          focusIO angularResultNoJs)
  104     b
