  101 newtype Handler a = Handler { unHandler :: ReaderT RequestContext IO a }
  102   deriving (Functor, Applicative, Alternative, Monad, MonadPlus, MonadIO, MonadBase IO, MonadThrow, MonadReader RequestContext)
  103 
  104 {-# INLINE runHandler #-}
  105 runHandler :: Handler a -> RequestContext -> IO a
  106 runHandler (Handler (ReaderT f)) = f
  107 
  108 instance MonadResource Handler where
  109   liftResourceT = focusIO . runInternalState
  110 
  111 instance MonadBaseControl IO Handler where
  112   type StM Handler a = a
  113   liftBaseWith f = Handler $ liftBaseWith $ \r -> f (r . unHandler)
  114   restoreM = Handler . restoreM
