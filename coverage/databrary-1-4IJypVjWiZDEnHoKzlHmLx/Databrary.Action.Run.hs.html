   45 -- |
   46 -- Transform a web request to a lower-level action.
   47 --
   48 -- Given an action that runs in a top-level Handler, build the necessary context
   49 -- for that action, and run in it in the base-level ContextM
   50 --
   51 -- Handler has a richer context: it has all of ActionContext, plus Identity and
   52 -- the Wai Request.
   53 withHandler
   54     :: forall a
   55      . Request -- ^ The wai request to handle
   56     -> Identity -- ^ The identity to use for this action
   57     -> Handler a -- ^ The action to perform
   58     -> ContextM a -- ^ The base-level control access to the system
   59 withHandler waiReq identity h =
   60     let (handler :: ReaderT RequestContext IO a) = unHandler h
   61     in withReaderT (\(c :: ActionContext) -> RequestContext c waiReq identity) handler
   62 
   63 -- | Authentication requirements for an 'Action'.
   64 data NeedsAuthentication = NeedsAuthentication | DoesntNeedAuthentication
   65 
   66 -- | This type captures both the authentication needs and the handler for a
   67 -- route.
   68 --
   69 -- This type is only found buried within an 'ActionRoute'.
   70 data Action = Action
   71     { _actionAuthentication :: !NeedsAuthentication
   72     , _actionM :: !(Handler Response)
   73     }
   74 
   75 -- | Special-purpose context for determing the user's identity. We don't need
   76 -- the full Handler for that, and since this is sensitive work, we don't want to
   77 -- just cheat and use it anyway.
   78 data IdContext = IdContext
   79     { ctxReq :: Wai.Request
   80     , ctxSec :: Secret
   81     , ctxConn :: DBConn
   82     }
   83 
   84 instance Has Wai.Request IdContext where view = ctxReq
   85 instance Has Secret IdContext where view = ctxSec
   86 instance Has DBConn IdContext where view = ctxConn

   88 -- | Run a Databrary 'Action' as a Wai Application.
   89 --
   90 -- This is the lowest level of the Databrary \"web framework\". It runs the
   91 -- requested action with some resolved identity to build the HTTP response.
   92 runAction
   93     :: Service -- ^ All the low-level system capabilities
   94     -> Action -- ^ Action to run
   95     -> Wai.Application -- ^ Callback for Wai
   96 runAction service (Action needsAuth act) waiReq waiSend
   97     = let
   98           ident' :: Secret -> DBConn -> ReaderT ActionContext IO Identity
   99           ident' sec con = case needsAuth of
  100               NeedsAuthentication -> runReaderT determineIdentity (IdContext waiReq sec con)
  101               DoesntNeedAuthentication -> return IdentityNotNeeded
  102           authenticatedAct :: ContextM (Identity, Response)
  103           authenticatedAct = do
  104               sec <- peek
  105               conn <- peek
  106               identity <- ident' sec conn
  107               waiResponse <-
  108                   ReaderT
  109                       -- runResult is IO Response -> IO Response
  110                       $ \ctx -> runResult $ runHandler
  111                             act
  112                             (RequestContext ctx waiReq identity)
  113               return (identity, waiResponse)
  114       in do
  115           ts <- getCurrentTime
  116           (identityUsed, waiResponse) <- runContextM authenticatedAct service
  117           logAccess
  118               ts
  119               waiReq
  120               (extractFromIdentifiedSessOrDefault
  121                   Nothing
  122                   (Just . (show :: Id Party -> String) . view)
  123                   identityUsed
  124               )
  125               waiResponse
  126               (serviceLogs service)
  127           let
  128               isdb = isDatabraryClient waiReq
  129               waiResponse' = Wai.mapResponseHeaders
  130                   (((hDate, formatHTTPTimestamp ts) :)
  131                   . (if isdb then ((hCacheControl, "no-cache") :) else id)
  132                   )
  133                   waiResponse
  134           waiSend $ if Wai.requestMethod waiReq == methodHead
  135               then emptyResponse
  136                   (Wai.responseStatus waiResponse')
  137                   (Wai.responseHeaders waiResponse')
  138               else waiResponse'

  140 -- | Run a Handler action in the background (IO).
  141 --
  142 -- A new ContextM is built, with a fresh guaranteed db connection and timestamp,
  143 -- via 'runContextM'
  144 forkAction
  145     :: Handler a -- ^ Handler action to run
  146     -> RequestContext
  147     -- ^ Original context
  148     --
  149     -- The background action inherits this context's 'Request' and 'Service'.
  150     -> (Either SomeException a -> IO ())
  151     -- ^ Cleanup to run when the action exits
  152     -> IO ThreadId
  153 forkAction h reqCtx = forkFinally $ runContextM (withHandler req ident h) srv
  154   where
  155     req = contextRequest reqCtx
  156     ident = requestIdentity reqCtx
  157     srv = contextService (requestContext reqCtx)

  159 -- | Tag an 'Action' as needing authentication.
  160 --
  161 -- For reference, note that Yesod's equivalent to this function returns
  162 -- @Handler Identity@, and throws a permission error when the user isn't
  163 -- identified.
  164 withAuth
  165     :: Handler Response -- ^ The handler for the route
  166     -> Action -- ^ The bundled action
  167 withAuth = Action NeedsAuthentication
  168 
  169 -- | Tag an 'Action' as not needing auth. See note at 'withAuth'.
  170 withoutAuth :: Handler Response -> Action
  171 withoutAuth = Action DoesntNeedAuthentication
  172 
  173 -- | This may be like a 'su' that allows running an action as a different
  174 -- identity.
  175 withReAuth
  176     :: SiteAuth -- ^ The identity to assume
  177     -> Handler a
  178     -- ^ The action to perform as the assumed identity
  179     --
  180     -- Note that one might argue for an indexed form of 'Handler' here, a la
  181     -- Selda and subqueries.
  182     -> Handler a -- ^ The re-authenticated action, packaged into the original context
  183 withReAuth u =
  184     Handler
  185         . local (\a -> a { requestIdentity = ReIdentified u })
  186         . unHandler
