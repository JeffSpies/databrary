   33 -- |
   34 -- Transform a web request to a lower-level action.
   35 --
   36 -- Given an action that runs in a top-level Handler, build the necessary context
   37 -- for that action, and run in it in the base-level ContextM
   38 --
   39 -- Handler has a richer context: it has all of ActionContext, plus Identity and
   40 -- the Wai Request.
   41 withHandler
   42     :: forall a
   43      . Request -- ^ The wai request to handle
   44     -> Identity -- ^ The identity to use for this action
   45     -> Handler a -- ^ The action to perform
   46     -> ContextM a -- ^ The base-level control access to the system
   47 withHandler waiReq identity h =
   48     let (handler :: ReaderT RequestContext IO a) = unHandler h
   49     in withReaderT (\(c :: ActionContext) -> RequestContext c waiReq identity) handler
   50 
   51 data NeedsAuth = NeedsAuth | DoesntNeedAuth
   52 
   53 data Action = Action
   54   { _actionAuth :: !NeedsAuth
   55   , _actionM :: !(Handler Response)
   56   }

   58 -- | Special-purpose context for determing the user's identity. We don't need
   59 -- the full Handler for that, and since this is sensitive work, we don't want to
   60 -- just cheat and use it anyway.
   61 data IdContext = IdContext
   62     { ctxReq :: Wai.Request
   63     , ctxSec :: Secret
   64     , ctxConn :: DBConn }
   65 
   66 instance Has Wai.Request IdContext where view = ctxReq
   67 instance Has Secret IdContext where view = ctxSec
   68 instance Has DBConn IdContext where view = ctxConn
   69 
   70 -- | Run the requested action with some resolved identity,
   71 -- send response, with some header decorations.
   72 runAction :: Service -> Action -> Wai.Application
   73 runAction service (Action needsAuth act) waiReq waiSend
   74     = let
   75           ident' :: Secret -> DBConn -> ReaderT ActionContext IO Identity
   76           ident' sec con = case needsAuth of
   77               NeedsAuth -> runReaderT determineIdentity (IdContext waiReq sec con)
   78               DoesntNeedAuth -> return IdentityNotNeeded
   79           authenticatedAct :: ContextM (Identity, Response)
   80           authenticatedAct = do
   81               sec <- peek
   82               conn <- peek
   83               identity <- ident' sec conn
   84               waiResponse <-
   85                   ReaderT
   86                       -- runResult is IO Response -> IO Response
   87                       $ \ctx -> runResult $ runHandler
   88                             act
   89                             (RequestContext ctx waiReq identity)
   90               return (identity, waiResponse)
   91       in do
   92           ts <- getCurrentTime
   93           (identityUsed, waiResponse) <- runContextM authenticatedAct service
   94           logAccess
   95               ts
   96               waiReq
   97               (extractFromIdentifiedSessOrDefault
   98                   Nothing
   99                   (Just . (show :: Id Party -> String) . view)
  100                   identityUsed
  101               )
  102               waiResponse
  103               (serviceLogs service)
  104           let
  105               isdb = isDatabraryClient waiReq
  106               waiResponse' = Wai.mapResponseHeaders
  107                   (((hDate, formatHTTPTimestamp ts) :)
  108                   . (if isdb then ((hCacheControl, "no-cache") :) else id)
  109                   )
  110                   waiResponse
  111           waiSend $ if Wai.requestMethod waiReq == methodHead
  112               then emptyResponse
  113                   (Wai.responseStatus waiResponse')
  114                   (Wai.responseHeaders waiResponse')
  115               else waiResponse'

  117 forkAction :: Handler a -> RequestContext -> (Either SomeException a -> IO ()) -> IO ThreadId
  118 forkAction f (RequestContext c r i) = forkFinally $
  119   runContextM (withHandler r i f) (contextService c)
  120 
  121 withAuth :: Handler Response -> Action
  122 withAuth = Action NeedsAuth
  123 
  124 withoutAuth :: Handler Response -> Action
  125 withoutAuth = Action DoesntNeedAuth

  127 -- | This may be like a 'su' that allows running an action as a different
  128 -- SiteAuth.
  129 --
  130 -- FIXME: It is a little annoying that Identified carries a session, while
  131 -- ReIdentified carries a SiteAuth.
  132 withReAuth :: SiteAuth -> Handler a -> Handler a
  133 withReAuth u =
  134     Handler
  135         . local (\a -> a { requestIdentity = ReIdentified u })
  136         . unHandler
