   36 selfAuthorize :: Party -> Authorize
   37 selfAuthorize p =
   38   Authorize (Authorization (if partyId (partyRow p) == partyId (partyRow nobodyParty) then minBound else maxBound) p p) Nothing

   40 lookupAuthorizedParents :: (MonadDB c m, MonadHasIdentity c m) => Party -> Maybe Permission -> m [Authorize]
   41 lookupAuthorizedParents child perm = do
   42   ident <- peek
   43   dbQuery $ maybe
   44     $(selectQuery (selectAuthorizeParent 'child 'ident) "$")
   45     (\p -> $(selectQuery (selectAuthorizeParent 'child 'ident) "$WHERE (expires IS NULL OR expires > CURRENT_TIMESTAMP) AND site >= ${p} AND member >= ${p} AND (site <> 'NONE' OR member <> 'NONE')"))
   46     perm

   48 lookupAuthorizedChildren :: (MonadDB c m, MonadHasIdentity c m) => Party -> Maybe Permission -> m [Authorize]

   54     perm

   56 -- | Attempt to find an authorization request or grant from the child party to the granting parent party
   57 lookupAuthorize :: MonadDB c m => Party -> Party -> m (Maybe Authorize)
   58 lookupAuthorize child parent =
   59   dbQuery1 $
   60       (\mkAuthorize' -> mkAuthorize' child parent)
   61           <$> $(selectQuery
   62                     authorizeRow
   63                     -- only include authorizations that either have no expiration or have not expired yet
   64                     "$WHERE authorize.child = ${partyId $ partyRow child} AND authorize.parent = ${partyId $ partyRow parent} AND (expires IS NULL OR expires > CURRENT_TIMESTAMP)")
   65 
   66 lookupAuthorizeParent :: (MonadDB c m, MonadHasIdentity c m) => Party -> Id Party -> m (Maybe Authorize)
   67 lookupAuthorizeParent child parent = do
   68   ident <- peek
   69   dbQuery1 $ $(selectQuery (selectAuthorizeParent 'child 'ident) "$WHERE authorize.parent = ${parent} AND (expires IS NULL OR expires > CURRENT_TIMESTAMP)")

   71 lookupAuthorization :: (MonadDB c m, MonadHasIdentity c m) => Party -> Party -> m Authorization
   72 lookupAuthorization child parent
   73   | partyId (partyRow child) == partyId (partyRow parent) = return $ authorization $ selfAuthorize child
   74   | otherwise = do
   75     auth <- peek
   76     if partyId (view auth) == partyId (partyRow child) && partyId (partyRow parent) == partyId (partyRow rootParty)
   77       then return $ Authorization (siteAccess auth) child parent
   78       else fromMaybe (Authorization mempty child parent) <$>
   79         dbQuery1 ((\a -> a child parent) <$> $(selectQuery authorizationRow "!$WHERE authorize_view.child = ${partyId $ partyRow child} AND authorize_view.parent = ${partyId $ partyRow parent}"))
   80 
   81 changeAuthorize :: (MonadAudit c m) => Authorize -> m ()
   82 changeAuthorize auth = do
   83   ident <- getAuditIdentity
   84   (r, _) <- updateOrInsert
   85     $(updateAuthorize 'ident 'auth)
   86     $(insertAuthorize 'ident 'auth)
   87   when (r /= 1) $ fail $ "changeAuthorize: " ++ show r ++ " rows"
   88 
   89 removeAuthorize :: (MonadAudit c m) => Authorize -> m Bool
   90 removeAuthorize auth = do
   91   ident <- getAuditIdentity
   92   dbExecute1 $(deleteAuthorize 'ident 'auth)
   93 
   94 authorizationActive :: Authorization -> Bool
   95 authorizationActive Authorization{ authorizeAccess = a } = a /= mempty
   96 
   97 authorizeExpired :: Authorize -> Timestamp -> Bool
   98 authorizeExpired Authorize{ authorizeExpires = Just e } = (e <)
   99 authorizeExpired _ = const False
  100 
  101 authorizeActive :: Authorize -> Timestamp -> Bool
  102 authorizeActive a t = authorizationActive (authorization a) && not (authorizeExpired a t)
  103 
  104 authorizeJSON :: JSON.ToObject o => Authorize -> o
  105 authorizeJSON Authorize{..} = accessJSON (authorizeAccess authorization)
  106   <> "expires" `JSON.kvObjectOrEmpty` authorizeExpires

  108 lookupAuthorizeActivity :: (MonadDB c m, MonadHasIdentity c m) => Int -> m [(Timestamp, Party)]
  109 lookupAuthorizeActivity limit = do
  110   ident :: Identity <- peek
  111   dbQuery $(selectQuery (selectAuthorizeActivity 'ident) "$JOIN authorize_view ON audit.parent = authorize_view.child AND authorize_view.parent = 0 WHERE audit.audit_action IN ('add','change') AND audit.site >= 'EDIT' AND authorize_view.site > 'EDIT' ORDER BY audit.audit_time DESC LIMIT ${fromIntegral limit :: Int64}")
