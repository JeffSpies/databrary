   24 type instance IdType Party = Int32
   25 
   26 data PartyRow = PartyRow
   27   { partyId :: Id Party
   28   , partySortName :: T.Text
   29   , partyPreName :: Maybe T.Text
   30   , partyORCID :: Maybe ORCID
   31   , partyAffiliation :: Maybe T.Text
   32   , partyURL :: Maybe URI
   33   } deriving (Eq, Show)
   34 
   35 -- | Represents users, institutions, labs, *and* groups.
   36 data Party = Party
   37   { partyRow :: !PartyRow
   38   , partyAccount :: Maybe Account
   39   -- , partySiteAccess :: Access -- site-level access this party is granted under root (currently SiteAuth only)
   40   , partyPermission :: Permission -- ^ permission current user has over this party
   41   , partyAccess :: Maybe Access -- ^ direct authorization this party has granted to current user
   42   }
   43 
   44 data Account = Account
   45   { accountEmail :: BS.ByteString
   46   , accountParty :: Party
   47   }
   48 
   49 -- makeHasRec ''PartyRow ['partyId]
   50 instance Has (Id Party) PartyRow where
   51   view = partyId
   52 -- makeHasRec ''Party ['partyRow]
   53 instance Has PartyRow Party where
   54   view = partyRow
   55 instance Has (Id Party) Party where
   56   view = (view . partyRow)
   57 -- makeHasRec ''Account ['accountParty]
   58 instance Has Party Account where
   59   view = accountParty
   60 instance Has PartyRow Account where
   61   view = (view . accountParty)
   62 instance Has (Id Party) Account where
   63   view = (view . accountParty)
   64 
   65 instance Has Access Party where
   66   view Party{ partyAccess = Just a } = a
   67   view _ = mempty
   68 
   69 instance Kinded Party where
   70   kindOf _ = "party"
   71 
   72 -- | TODO: clarify. This is not necessarily a session, but... some user (human
   73 -- being) who has been granted access to the site. There is a corner case
   74 -- indirection because sometimes a job runs a human being.
   75 data SiteAuth = SiteAuth
   76   { siteAccount :: Account -- ^ maybe should be Party (for nobody)
   77   , accountPasswd :: Maybe BS.ByteString
   78   , siteAccess :: Access -- ^ Still figuring out what an 'Access' is.
   79   }
   80 
   81 -- makeHasRec ''SiteAuth ['siteAccount, 'siteAccess]
   82 instance Has Account SiteAuth where
   83   view = siteAccount
   84 instance Has Party SiteAuth where
   85   view = (view . siteAccount)
   86 instance Has PartyRow SiteAuth where
   87   view = (view . siteAccount)
   88 instance Has (Id Party) SiteAuth where
   89   view = (view . siteAccount)
   90 instance Has Access SiteAuth where
   91   view = siteAccess
   92 
   93 deriveLiftMany [''PartyRow, ''Party, ''Account]

   96 nobodySiteAuth :: SiteAuth

  100     , accountParty = Party

  105         , partyORCID = Nothing
  106         , partyAffiliation = Nothing
  107         , partyURL = Nothing

  109       , partyAccount = Nothing
  110       , partyPermission = PermissionREAD
  111       , partyAccess = Just minBound

  114   , accountPasswd = Nothing

  118 -- | Uninitialized Party object to be used in creating new parties (and accounts)
  119 blankParty :: Party
  120 blankParty = Party
  121   { partyRow = PartyRow
  122     { partyId = error "blankParty"
  123     , partySortName = ""
  124     , partyPreName = Nothing
  125     , partyORCID = Nothing
  126     , partyAffiliation = Nothing
  127     , partyURL = Nothing
  128     }
  129   , partyAccount = Nothing
  130   , partyPermission = PermissionNONE
  131   , partyAccess = Nothing
  132   }
  133 
  134 -- | Uninitialized Account object to be used in creating new accounts
  135 blankAccount :: Account
  136 blankAccount = Account
  137   { accountParty = blankParty{ partyAccount = Just blankAccount }
  138   , accountEmail = error "blankAccount"
  139   }
