   13 import Control.Exception (Exception, throwIO, throw, handle)
   14 import Control.Monad (join)
   15 import Control.Monad.IO.Class (MonadIO, liftIO)
   16 import qualified Data.ByteString as BS
   17 import qualified Data.ByteString.Builder as BSB

   19 import qualified Data.Text as T
   20 import qualified Data.Text.Encoding as TE
   21 import qualified Data.Text.Lazy as TL
   22 import qualified Data.Text.Lazy.Encoding as TLE
   23 import Data.Typeable (Typeable)
   24 import qualified Network.HTTP.Client as HC
   25 import Network.HTTP.Types (ResponseHeaders, Status, ok200, hContentType)
   26 import Network.Wai (Response, responseBuilder, responseLBS, StreamingBody, responseStream, FilePart(..), responseFile, responseStatus)
   27 import System.Posix.Types (FileOffset)
   28 import qualified Text.Blaze.Html as Html
   29 import qualified Text.Blaze.Html.Renderer.Utf8 as Html
   30 import qualified Conduit as CND
   31 import Conduit (Source, (.|))

   33 
   34 import qualified Databrary.JSON as JSON
   35 
   36 class ResponseData r where
   37   response :: Status -> ResponseHeaders -> r -> Response
   38 
   39 instance ResponseData (Status -> ResponseHeaders -> Response) where
   40   response s h r = r s h
   41 
   42 instance ResponseData BSB.Builder where
   43   response = responseBuilder
   44 
   45 instance ResponseData BSL.ByteString where
   46   response = responseLBS
   47 
   48 instance ResponseData BS.ByteString where
   49   response s h = responseBuilder s h . BSB.byteString
   50 
   51 instance ResponseData (Source (CND.ResourceT IO) BS.ByteString) where
   52   response s h src =
   53     responseStream s h
   54       (\send _ -> do
   55          CND.runConduitRes (src .| (CND.mapM_C (\bs -> CND.lift (send (DBB.fromByteString bs))))))
   56 
   57 instance ResponseData StreamingBody where
   58   response = responseStream
   59 
   60 instance ResponseData ((BSB.Builder -> IO ()) -> IO ()) where
   61   response s h f = responseStream s h (\w _ -> f w)
   62 
   63 instance ResponseData ((BS.ByteString -> IO ()) -> IO ()) where
   64   response s h f = responseStream s h (\w l -> f (\b -> if BS.null b then l else w (BSB.byteString b)))
   65 
   66 instance ResponseData (FilePath, Maybe FilePart) where
   67   response s h (f, p) = responseFile s h f p
   68 
   69 instance ResponseData (FilePath, FilePart) where
   70   response s h (f, p) = response s h (f, Just p)
   71 
   72 instance ResponseData (FilePath, Maybe FileOffset) where
   73   response s h (f, z) = response s h (f, join (FilePart 0) . toInteger <$> z)
   74 
   75 instance ResponseData String where
   76   response s h =
   77     response s ((hContentType, "text/plain;charset=utf-8") : h) . BSB.stringUtf8
   78 
   79 instance ResponseData T.Text where
   80   response s h =
   81     response s ((hContentType, "text/plain;charset=utf-8") : h) . TE.encodeUtf8Builder
   82 
   83 instance ResponseData TL.Text where
   84   response s h =
   85     response s ((hContentType, "text/plain;charset=utf-8") : h) . TLE.encodeUtf8Builder
   86 
   87 instance ResponseData JSON.Value where
   88   response s h =
   89     response s ((hContentType, "application/json") : h) . JSON.encode
   90 
   91 instance ResponseData JSON.Encoding where
   92   response s h =
   93     response s ((hContentType, "application/json") : h) . JSON.fromEncoding
   94 
   95 instance ResponseData JSON.Series where
   96   response s h =
   97     response s h . JSON.pairs
   98 
   99 instance (JSON.ToJSON k, JSON.ToObject o, ResponseData o) => ResponseData (JSON.Record k o) where
  100   response s h =
  101     response s h . JSON.recordObject
  102 
  103 instance ResponseData Html.Html where
  104   response s h =
  105     response s ((hContentType, "text/html;charset=utf-8") : h) . Html.renderHtmlBuilder
  106 
  107 emptyResponse :: Status -> ResponseHeaders -> Response
  108 emptyResponse s h = response s h BS.empty
  109 
  110 okResponse :: ResponseData r => ResponseHeaders -> r -> Response
  111 okResponse = response ok200
  112 
  113 newtype Result = Result { resultResponse :: Response } deriving (Typeable)
  114 instance Show Result where
  115   showsPrec p (Result r) = showParen (p > 10)
  116     $ showString "Result " . showsPrec 11 (responseStatus r)
  117 instance Exception Result
  118 
  119 result :: MonadIO m => Response -> m a
  120 result = liftIO . throwIO . Result
  121 
  122 unsafeResult :: Response -> a
  123 unsafeResult = throw . Result
  124 
  125 runResult :: IO Response -> IO Response
  126 runResult = handle (return . resultResponse)
  127 
  128 proxyResponse :: HC.Response BSL.ByteString -> Response
  129 proxyResponse r = responseLBS
  130   (HC.responseStatus r)
  131   (filter ((/= "transfer-encoding") . fst) $ HC.responseHeaders r)
  132   (HC.responseBody r)
