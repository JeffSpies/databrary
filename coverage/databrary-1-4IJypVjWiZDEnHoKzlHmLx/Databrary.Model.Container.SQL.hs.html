1 {-# LANGUAGE TemplateHaskell, OverloadedStrings #-}
    2 module Databrary.Model.Container.SQL
    3   ( selectContainerRow
    4   , selectVolumeContainer
    5   , selectContainer
    6   , insertContainer
    7   , updateContainer
    8   , deleteContainer
    9   -- for expanded query
   10   , setContainerId
   11   ) where
   12 
   13 import qualified Language.Haskell.TH as TH
   14 
   15 import Databrary.Model.SQL.Select
   16 import Databrary.Model.Id.Types
   17 import Databrary.Model.Audit.SQL
   18 import Databrary.Model.Volume.SQL
   19 import Databrary.Model.Release.SQL
   20 import Databrary.Model.Container.Types
   21 
   22 selectContainerRow :: Selector -- ^ @'ContainerRow'@
   23 selectContainerRow = selectColumns 'ContainerRow "container" ["id", "top", "name", "date"]
   24 
   25 selectVolumeContainer :: Selector -- ^ @'Volume' -> 'Container'@
   26 selectVolumeContainer = selectJoin 'Container
   27   [ selectContainerRow
   28   , maybeJoinOn "container.id = slot_release.container AND slot_release.segment = '(,)'"
   29     releaseRow
   30   ]
   31 
   32 selectContainer :: TH.Name -- ^ @'Identity'@
   33   -> Selector -- ^ @'Container'@
   34 selectContainer ident = selectJoin '($)
   35   [ selectVolumeContainer
   36   , joinOn "container.volume = volume.id" $ selectVolume ident
   37   ]
   38 
   39 containerKeys :: String -- ^ @'Container'@
   40   -> [(String, String)]
   41 containerKeys o =
   42   [ ("id", "${containerId $ containerRow " ++ o ++ "}") ]
   43 
   44 containerSets :: String -- ^ @'Container'@
   45   -> [(String, String)]
   46 containerSets o =
   47   [ ("volume", "${volumeId $ volumeRow $ containerVolume " ++ o ++ "}")
   48   , ("top", "${containerTop $ containerRow " ++ o ++ "}")
   49   , ("name", "${containerName $ containerRow " ++ o ++ "}")
   50   , ("date", "${containerDate $ containerRow " ++ o ++ "}")
   51   ]
   52 
   53 setContainerId :: Container -> Id Container -> Container
   54 setContainerId o i = o{ containerRow = (containerRow o) { containerId = i } }
   55 
   56 insertContainer :: TH.Name -- ^ @'AuditIdentity'@
   57   -> TH.Name -- ^ @'Container'@
   58   -> TH.ExpQ -- ^ @'Container'@
   59 insertContainer ident o = auditInsert ident "container"
   60   (containerSets (nameRef o))
   61   (Just $ selectOutput $ selectMap ((TH.VarE 'setContainerId `TH.AppE` TH.VarE o) `TH.AppE`) $ selectColumn "container" "id")
   62 
   63 updateContainer :: TH.Name -- ^ @'AuditIdentity'@
   64   -> TH.Name -- ^ @'Container'@
   65   -> TH.ExpQ -- ^ @()@
   66 updateContainer ident o = auditUpdate ident "container"
   67   (containerSets (nameRef o))
   68   (whereEq $ containerKeys (nameRef o))
   69   Nothing
   70 
   71 deleteContainer :: TH.Name -- ^ @'AuditIdentity'@
   72   -> TH.Name -- ^ @'Container'@
   73   -> TH.ExpQ -- ^ @()@
   74 deleteContainer ident o = auditDelete ident "container"
   75   (whereEq $ containerKeys (nameRef o))
   76   Nothing
