1 {-# LANGUAGE OverloadedStrings, TemplateHaskell #-}
    2 module Databrary.Model.Token.SQL
    3   ( selectLoginToken
    4   , selectSession
    5   , selectUpload
    6   -- for expanded queries
    7   , makeUpload
    8   ) where
    9 
   10 import qualified Data.ByteString as BS
   11 import Data.Int (Int64)
   12 
   13 import Databrary.Model.SQL.Select
   14 import Databrary.Model.Party.Types
   15 import Databrary.Model.Party.SQL (selectSiteAuth)
   16 import Databrary.Model.Token.Types
   17 
   18 tokenRow :: String -- ^ table name
   19   -> Selector -- ^ @'Token'@
   20 tokenRow table = selectColumns 'Token table ["token", "expires"]
   21 
   22 accountTokenRow :: String -- ^ table name
   23   -> Selector -- ^ @'AccountToken'@
   24 accountTokenRow table = selectJoin 'AccountToken
   25   [ tokenRow table
   26   , joinOn (table ++ ".account = account.id") selectSiteAuth
   27   ]
   28 
   29 selectLoginToken :: Selector -- @'Session'@
   30 selectLoginToken =
   31   addSelects 'LoginToken (accountTokenRow "login_token") [SelectColumn "login_token" "password"]
   32 
   33 selectSession :: Selector -- @'Session'@
   34 selectSession =
   35   addSelects 'Session (accountTokenRow "session") [SelectColumn "session" "verf", SelectColumn "session" "superuser"]
   36 
   37 makeUpload :: Token -> BS.ByteString -> Int64 -> SiteAuth -> Upload
   38 makeUpload t n z u = Upload (AccountToken t u) n z
   39 
   40 selectUpload :: Selector -- @'SiteAuth' -> 'Upload'@
   41 selectUpload =
   42   addSelects 'makeUpload (tokenRow "upload") [SelectColumn "upload" "filename", SelectColumn "upload" "size"]
