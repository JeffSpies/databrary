   28 type ActionRoute a = R.RouteAction a Action
   29 
   30 actionURL :: Maybe Request -> R.RouteAction r a -> r -> Query -> BSB.Builder
   31 actionURL req r a q
   32   | R.requestMethod rr == GET = routeURL req rr q
   33   | otherwise = error $ "actionURL: " ++ show rr
   34   where rr = R.requestActionRoute r a
   35 
   36 actionURI :: Maybe Request -> R.RouteAction r a -> r -> Query -> URI
   37 actionURI req r a q
   38   | R.requestMethod rr == GET = routeURI req rr q
   39   | otherwise = error $ "actionURI: " ++ show rr
   40   where rr = R.requestActionRoute r a
   41 
   42 actionMethod :: R.RouteAction r a -> r -> Method
   43 actionMethod r = R.requestMethod . R.requestActionRoute r
   44 
   45 action :: Method -> PathParser r -> (r -> a) -> R.RouteAction r a
   46 action m p a = R.routePath p R.>* R.routeMethod m `R.RouteAction` a
   47 
   48 multipartAction :: R.RouteAction q a -> R.RouteAction q a
   49 multipartAction (R.RouteAction r a) =
   50   R.RouteAction (r R.>* (R.routeAccept "multipart/form-data" R.>| R.unit)) a
   51 
   52 data API
   53   = HTML
   54   | JSON
   55   deriving (Eq)
   56 
   57 pathHTML :: PathParser ()
   58 pathHTML = R.unit
   59 
   60 pathJSON :: PathParser ()
   61 pathJSON = "api"
   62 
   63 pathAPI :: PathParser API
   64 pathAPI = [I.biCase|Left () <-> JSON ; Right () <-> HTML|] R.>$< (pathJSON R.>|< pathHTML)
